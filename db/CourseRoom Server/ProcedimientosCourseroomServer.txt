DELIMITER //
CREATE PROCEDURE Obtener_Peticion()
BEGIN
  select * from Tb_Peticiones order by Fecha desc limit 10;
END //

DELIMITER //
CREATE PROCEDURE Obtener_Respuesta()
BEGIN
  select * from Tb_Respuestas order by Fecha desc limit 10;
END //
